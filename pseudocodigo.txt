        retorna altura do no

        se no for falso:
            retorna 0
        retorna função obter altura(esquerda do no) - função obter altura(diretia do no)
    
    função rotação da direita(y):
        x = esquerda de y
        T2 = direita de x

        direita de x = y
        esquerda de y = T2

        altura de y = 1 + função valor maximo da função obter altura(esquerda de y), obter altura(direita de y) 
        altura de x = 1 + função valor maximo da função obter altura(esquerda de x), obter altura(direita de x) 

        retorna x

    função rotação da esquerda(x):
        y = direita de x
        T2 = esquerda de y

        esquerda de y = x
        direita de x = T2

        altura de x = 1 + função valor maximo da função obter altura(esquerda de x), obter altura(direita de x) 
        altura de y = 1 + função valor maximo da função obter altura(esquerda de y), obter altura(direita de y) 

        retorna y
    
    função inserir(no, valor):
        se no for falso:
            retorna classe No(valor)
        
        se valor for < valor do no:
            esquerda do no = função inserir(esquerda do no, valor)
        senão se valor for > valor do no:
            direita do no = função inserir(direita do no, valor)
        senão:
            retorna no
        
        altura do no = 1 + função valor maximo da função obter altura(esquerda do no), função obter altura(direita do no)

        fator balanceado = função obter fator balanceamento(no)    

        se fator balanceado for > que 1 e valor for < que valor da esquerda do no:
            retorna função rotação da direita(no)
        se fator balanceado for < que -1 e valor for > que valor da direita do no:
            retorna função rotação da esquerda(no)
        se fator balanceado for > que 1 e valor for < que valor da esquerda do no:
            esquerda do no = função rotação da esquerda(esquerda do no)
            retorna função rotação da direita(no)
        se fator balanceado for < que -1 e valor for < que valor da direita do no:
            direita do no = função rotação da direita(direita do no)
            retorna rotação da esquerda(no)

        retorna no 

    função imprimir arvore(no, nivel=0, prefixo="Raiz: "):

                se esquerda do no for verdadeiro:

                senão:
                    printe(" " * ((nivel + 1) * 4) + "E: Vazio")
                se direita do no for verdadeiro:
                    função imprimir arvore(direita do no, nivel + 1, "D: ")
                senão:
                    printe(" " * ((nivel + 1) * 4) + "D: Vazio")
